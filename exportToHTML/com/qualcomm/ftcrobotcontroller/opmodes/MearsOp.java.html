72           //float throttle = -Test;

74           float right = gamepad1.right_stick_y; 
75           float left = gamepad1.left_stick_y; 
76           float arm = -gamepad2.right_stick_y; 
77    
78           // clip the right/left values so that the values never exceed +/- 1 
79           right = Range.clip(right, -1, 1);

81           arm = Range.clip(arm, -1, 1); 
82           // scale the joystick value to make it easier to control 
83           // the robot more precisely at slower speeds. 
84           right = (float) scaleInput(right); 
85           left = (float) scaleInput(left); 
86           arm = (float) scaleInput(arm); 
87           // write the values to the motors 
88           right_motor.setPower(right); 
89           left_motor.setPower(left); 
90           arm_motor.setPower(arm);

111      /* 
112       * This method scales the joystick input so for low joystick values, the 
113       * scaled value is less than linear.  This is to make it easier to drive 
114       * the robot more precisely at slower speeds. 
115       */ 
116      double scaleInput(double dVal)  { 
117          double[] scaleArray = { 0.0, 0.05, 0.09, 0.10, 0.12, 0.15, 0.18, 0.24, 
118                  0.30, 0.36, 0.43, 0.50, 0.60, 0.72, 0.85, 1.00, 1.00 }; 
119   
120          // get the corresponding index for the scaleInput array. 
121          int index = (int) (dVal * 16.0); 
122          if (index < 0) { 
123              index = -index; 
124          } else if (index > 16) { 
125              index = 16; 
126          } 
127   
128          double dScale = 0.0; 
129          if (dVal < 0) { 
130              dScale = -scaleArray[index]; 
131          } else { 
132              dScale = scaleArray[index]; 
133          } 
134   
135          return dScale; 
136      }
